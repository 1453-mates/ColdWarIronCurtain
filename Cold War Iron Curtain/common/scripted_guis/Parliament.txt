		context_type = player_context

		window_name = "legislation_menu_button"

		visible = {
			always = yes
		}	
		effects = {
			legislation_menu_open_click = {
				set_temp_variable = { show = 1 }
				subtract_from_temp_variable = { show = legislation_menu_show }

			}	
		}
	}
		
	legislation_menu = {

		context_type = player_context

		window_name = "legislation_menu"

		visible = {
			check_variable = { legislation_menu_show = 1 }
		}	
		effects = {
			policy_switch_click = {
				set_variable = { policy_legislation = 1 }
			}
			legislation_switch_click = {
				set_variable = { policy_legislation = 0 }
			}
			legislation_anti_racial_discrimination_click = {
				set_variable = { legislation_support = 0 }
				add_to_variable = { legislation_support = communism_seats_parliament2 }
				add_to_variable = { legislation_support = socialist_seats_parliament2 }
				add_to_variable = { legislation_support = Social_Democratic_seats_parliament2 }
				add_to_variable = { legislation_support = trotskyism_seats_parliament2 }
				add_to_variable = { legislation_support = democratic_seats_parliament2 }
				
				add_to_variable = { legislation_support_upper = communism_seats_parliament3 }
				add_to_variable = { legislation_support_upper = socialist_seats_parliament3 }
				add_to_variable = { legislation_support_upper = Social_Democratic_seats_parliament3 }
				add_to_variable = { legislation_support_upper = trotskyism_seats_parliament3 }
				add_to_variable = { legislation_support_upper = democratic_seats_parliament3 }
				
				set_country_flag = legislation_anti_racial_discrimination_vote
				set_country_flag = vote_in_progress
				activate_mission = Anti_Racial_Discrimination_Legislation


			}
		}
		triggers = {
			policy_switch_click_enabled = {
				check_variable = { policy_legislation = 0 }
			}
			legislation_switch_click_enabled = {
				check_variable = { policy_legislation = 1 }
			}
			legislation_anti_racial_discrimination_visible = {
				check_variable = { policy_legislation = 0 }
			}
			legislation_anti_racial_discrimination_click_enabled = {
				NOT = { has_country_flag = vote_in_progress }
				NOT = { has_idea = Anti_Racial_Discrimination }
				
			}
		}
	}
