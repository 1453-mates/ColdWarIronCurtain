fossil_fuel_pulse = {
	if = {
		limit = { 
			NOT = { check_variable = { fossil_fuel_oil_consumption = 0 } }
		}
		multiply_variable = { fossil_fuel_oil_consumption = -1 }  
		meta_effect = {
			text = {
				add_resource = {
					type = oil
					amount = [AMNT]
				}
			}
			AMNT = "[?fossil_fuel_oil_consumption]"
		}	
	}
	if = {
		limit = { 
			NOT = { check_variable = { fossil_fuel_elec_prod = 0 } }
		}
		multiply_variable = { fossil_fuel_elec_prod = -1 }  
		meta_effect = {
			text = {
				add_resource = {
					type = electricity
					amount = [AMNT]
				}
			}
			AMNT = "[?fossil_fuel_elec_prod]"
		}	
	}
	#fossil_fuel_powerplant
	add_to_variable = { fossil_fuel_oil_consumption = -1 }
	add_to_variable = { fossil_fuel_elec_prod = 5 }

	if = {
		limit = {
			controller = {
				has_tech = gas_I
			}
		}
		add_to_variable = { fossil_fuel_oil_consumption = 1 }  
	}
	if = {
		limit = {
			controller = {
				has_tech = gas_II
			}
		}
		add_to_variable = { fossil_fuel_oil_consumption = 1 }  
	}
	if = {
		limit = {
			controller = {
				has_tech = gas_III
			}
		}
		add_to_variable = { fossil_fuel_oil_consumption = 1 }  
	}	
	if = {
		limit = {
			controller = {
				has_tech = software_VI
			}
		}
		add_to_variable = { fossil_fuel_oil_consumption = 1 }  
	}	
	if = {

			controller = {
