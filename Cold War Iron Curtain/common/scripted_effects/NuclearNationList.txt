nuclear_power_array = {
	if = {
		limit = {
			AND = {
				has_idea = armed_nation_nuke
				NOT = {
					has_idea = Denuclearization
					is_in_array = {
						array = nuclear_nations
						value = 1
					}
				}
			}
		}
		add_to_array = {
			array = nuclear_nations
			value = 1
		}
	}
	if = {
		limit = {
			OR = {
				has_idea = Denuclearization
				has_idea = disarmed_nation_nuke
			}
		}
		remove_from_array = {
			array = nuclear_nations
			value = 1
		}
	}
}
